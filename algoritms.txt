{
    Type T sequence of T0;
    Type stack S1[1..n] of T0;
    Type stack S2[1..n] of T0;
}

shuffle(S1, S2)
    while top[S1] > 0 OR top[S2] > 0 do

        if top[S1] > 0 
            T[tail[T]+1] <- top[S1]
            top[S1] = top[S1]-1

        if top[S2] > 0
            T[tail[T]+1] <- top[S2]
            top[S2] = top[S2]-1
    
    return T;
